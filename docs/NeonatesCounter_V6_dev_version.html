 22class ImagePatcher:
 23    """
 24    Handles image padding and patch extraction for tiled inference.
 25
 26    Attributes:
 27        patch_size (int): Size of each patch.
 28        stride (int): Stride for patch extraction.
 29    """
 30    def __init__(self, patch_size=416, stride=376):
 31        """
 32        Initialize the ImagePatcher.
 33
 34        Args:
 35            patch_size (int): Size of each patch.
 36            stride (int): Stride for patch extraction.
 37        """
 38        self.patch_size = patch_size
 39        self.stride = stride
 40
 41    def pad_image(self, image):
 42        """
 43        Pad the image so its dimensions are multiples of the stride.
 44
 45        Args:
 46            image (np.ndarray): Input image.
 47
 48        Returns:
 49            np.ndarray: Padded image.
 50        """
 51        h, w, _ = image.shape
 52        pad_h = (self.stride - (h % self.stride)) if (h % self.stride) != 0 else 0
 53        pad_w = (self.stride - (w % self.stride)) if (w % self.stride) != 0 else 0
 54        return cv2.copyMakeBorder(image, 0, pad_h, 0, pad_w, cv2.BORDER_CONSTANT, value=(0, 0, 0))
 55
 56    def create_patches(self, image):
 57        """
 58        Create patches from the input image.
 59
 60        Args:
 61            image (np.ndarray): Input image.
 62
 63        Returns:
 64            np.ndarray: Array of patches.
 65        """
 66        return patchify(image, (self.patch_size, self.patch_size, 3), step=self.stride)

 97    """
 98    Reconstructs images from patches if needed.

115    def repatch_image(self, patches, image_shape):
116        """
117        Reconstruct the image from patches.
118
119        Args:
120            patches (np.ndarray): Array of patches.
121            image_shape (tuple): Shape of the original image.
122
123        Returns:
124            np.ndarray: Reconstructed image.
125        """
126        h, w, c = image_shape
127        reconstructed = np.zeros((h, w, c), dtype=np.float32)
128        count_map = np.zeros((h, w, c), dtype=np.float32)
129
130        for i in range(patches.shape[0]):
131            for j in range(patches.shape[1]):
132                patch = patches[i, j, 0]
133                y_start, x_start = i * self.stride, j * self.stride
134                y_end, x_end = y_start + self.patch_size, x_start + self.patch_size
135                reconstructed[y_start:y_end, x_start:x_end] += patch
136                count_map[y_start:y_end, x_start:x_end] += 1
137
138        count_map[count_map == 0] = 1
139        return (reconstructed / count_map).astype(np.uint8)

208    def load_ground_truth(self):
209        """
210        Load ground truth from VGG Image Annotator JSON file.
211
212        Returns:
213            dict: Mapping from image stem to list of ground truth boxes.
214        """
215        gt_data = {}
216        if not os.path.exists(self.annotation_path):
217            print(f"Warning: Annotation file not found at {self.annotation_path}")
218            return gt_data
219        try:
220            with open(self.annotation_path, 'r') as f:
221                via_data = json.load(f)
222            for img_id_key, img_info in via_data.items():
223                filename = img_info.get('filename')
224                if not filename:
225                    continue
226                image_stem = Path(filename).stem
227                gt_boxes = []
228                for region in img_info.get('regions', []):
229                    shape_attributes = region.get('shape_attributes', {})
230                    region_attributes = region.get('region_attributes', {})
231                    if (shape_attributes.get('name') == 'rect' and 
232                        region_attributes.get('type') == '0'):
233                        x = shape_attributes.get('x')
234                        y = shape_attributes.get('y')
235                        width = shape_attributes.get('width')
236                        height = shape_attributes.get('height')
237                        if all(v is not None for v in [x, y, width, height]):
238                            x1, y1 = int(x), int(y)
239                            x2, y2 = int(x + width), int(y + height)
240                            gt_boxes.append([x1, y1, x2, y2])
241                if gt_boxes:
242                    gt_data[image_stem] = gt_boxes
243            print(f"Loaded ground truth for {len(gt_data)} images from {self.annotation_path}")
244        except Exception as e:
245            print(f"Error loading ground truth: {e}")
246        return gt_data

410    def deduplicate_boxes(self, boxes_with_scores, iou_threshold=0.45, return_scores=False):
411        """
412        Remove duplicate bounding boxes using Non-Maximum Suppression (NMS).
413
414        Args:
415            boxes_with_scores (list): List of boxes with scores.
416            iou_threshold (float): IoU threshold for NMS.
417            return_scores (bool): Whether to return scores.
418
419        Returns:
420            list: Deduplicated boxes.
421        """
422        if len(boxes_with_scores) == 0:
423            return []
424        boxes_np = np.array([[x1, y1, x2, y2] for x1, y1, x2, y2, _ in boxes_with_scores])
425        scores = np.array([score for *_, score in boxes_with_scores])
426        boxes_tensor = torch.tensor(boxes_np, dtype=torch.float32)
427        scores_tensor = torch.tensor(scores, dtype=torch.float32)
428        keep_indices = nms(boxes_tensor, scores_tensor, iou_threshold)
429        kept = np.array(boxes_with_scores)[keep_indices]
430        if return_scores:
431            return [(int(x1), int(y1), int(x2), int(y2), float(score)) for x1, y1, x2, y2, score in kept]
432        else:
433            return [(int(x1), int(y1), int(x2), int(y2)) for x1, y1, x2, y2, _ in kept]

435    def filter_by_shape_metrics_percentile(self, image, detections):
436        """
437        Filter detections by shape metrics using percentile thresholds calculated
438        from all detected contours in the image.
439
440        Args:
441            image (np.ndarray): Input image.
442            detections (list): List of detection boxes.
443
444        Returns:
445            list: Filtered detections.
446        """
447        circularities = []
448        solidities = []
449        areas = []
450        extents = []
451        convexities = []
452        contour_data = []
453        for det in detections:
454            x1, y1, x2, y2, score = det[:5]
455            roi = image[y1:y2, x1:x2]
456            if roi.size == 0:
457                continue
458            gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
459            contours = self._find_contours(gray)
460            for cnt in contours:
461                area = cv2.contourArea(cnt)
462                if area == 0:
463                    continue
464                perimeter = cv2.arcLength(cnt, True)
465                circularity = 4 * np.pi * area / (perimeter ** 2) if perimeter > 0 else 0
466                hull = cv2.convexHull(cnt)
467                hull_area = cv2.contourArea(hull)
468                solidity = float(area) / hull_area if hull_area > 0 else 0
469                x, y, w, h = cv2.boundingRect(cnt)
470                rect_area = w * h
471                extent = float(area) / rect_area if rect_area > 0 else 0
472                convexity = float(area) / cv2.contourArea(hull) if cv2.contourArea(hull) > 0 else 0
473                circularities.append(circularity)
474                solidities.append(solidity)
475                areas.append(area)
476                extents.append(extent)
477                convexities.append(convexity)
478                contour_data.append({
479                    "det": det,
480                    "circularity": circularity,
481                    "solidity": solidity,
482                    "area": area,
483                    "extent": extent,
484                    "convexity": convexity
485                })
486        if not contour_data:
487            return []
488        min_circ_thresh = np.percentile(circularities, 1)
489        max_circ_thresh = np.percentile(circularities, 100)
490        min_solid_thresh = np.percentile(solidities, 0)
491        max_solid_thresh = np.percentile(solidities, 100)
492        min_area_thresh = np.percentile(areas, 3)
493        max_area_thresh = np.percentile(areas, 100)
494        min_extent_thresh = np.percentile(extents, 1)
495        max_extent_thresh = np.percentile(extents, 100)
496        min_convex_thresh = np.percentile(convexities, 0)
497        max_convex_thresh = np.percentile(convexities, 100)
498        filtered_detections = []
499        det_seen = set()
500        for cdata in contour_data:
501            det = cdata["det"]
502            det_tuple = tuple(det)
503            if det_tuple in det_seen:
504                continue
505            if (0.1 <= cdata["circularity"] <= max_circ_thresh and
506                0.2 <= cdata["solidity"] <= max_solid_thresh and
507                5 <= cdata["area"] <= max_area_thresh and
508                0.1 <= cdata["extent"] <= max_extent_thresh and
509                0.1 <= cdata["convexity"] <= max_convex_thresh):
510                filtered_detections.append(det)
511                det_seen.add(det_tuple)
512        return filtered_detections

528    def _calculate_iou(self, boxA, boxB):
529        """
530        Calculates Intersection over Union (IoU) of two bounding boxes.
531
532        Args:
533            boxA (list): First box [x1, y1, x2, y2].
534            boxB (list): Second box [x1, y1, x2, y2].
535
536        Returns:
537            float: IoU value.
538        """
539        xA = max(boxA[0], boxB[0])
540        yA = max(boxA[1], boxB[1])
541        xB = min(boxA[2], boxB[2])
542        yB = min(boxA[3], boxB[3])
543        interArea = max(0, xB - xA + 1) * max(0, yB - yA + 1)
544        boxAArea = (boxA[2] - boxA[0] + 1) * (boxA[3] - boxA[1] + 1)
545        boxBArea = (boxB[2] - boxB[0] + 1) * (boxB[3] - boxB[1] + 1)
546        iou = interArea / float(boxAArea + boxBArea - interArea)
547        return iou

549    def calculate_metrics(self, all_gt_boxes, all_det_boxes, iou_threshold=0.3):
550        """
551        Calculates Precision and Recall across all images.
552
553        Args:
554            all_gt_boxes (list): List of ground truth boxes.
555            all_det_boxes (list): List of detection boxes.
556            iou_threshold (float): IoU threshold for matching.
557
558        Returns:
559            tuple: (precision, recall, true_positives, false_positives, false_negatives)
560        """
561        all_det_boxes.sort(key=lambda x: x[4], reverse=True)
562        true_positives = 0
563        false_positives = 0
564        false_negatives = 0
565        matched_gt = [False] * len(all_gt_boxes)
566        for det_idx, det_box_with_conf in enumerate(all_det_boxes):
567            det_box = det_box_with_conf[:4]
568            best_iou = 0
569            best_gt_idx = -1
570            for gt_idx, gt_box in enumerate(all_gt_boxes):
571                iou = self._calculate_iou(det_box, gt_box)
572                if iou > best_iou:
573                    best_iou = iou
574                    best_gt_idx = gt_idx
575            if best_iou >= iou_threshold:
576                if not matched_gt[best_gt_idx]:
577                    true_positives += 1
578                    matched_gt[best_gt_idx] = True
579                else:
580                    false_positives += 1 
581            else:
582                false_positives += 1
583        false_negatives = sum(1 for matched in matched_gt if not matched)
584        precision = true_positives / (true_positives + false_positives) if (true_positives + false_positives) > 0 else 0
585        recall = true_positives / (true_positives + false_negatives) if (true_positives + false_negatives) > 0 else 0
586        return precision, recall, true_positives, false_positives, false_negatives

436    def filter_by_shape_metrics_percentile(self, image, detections):
437        """
438        Filter detections by shape metrics using percentile thresholds calculated
439        from all detected contours in the image.
440
441        Args:
442            image (np.ndarray): Input image.
443            detections (list): List of detection boxes.
444
445        Returns:
446            list: Filtered detections.
447        """
448        circularities = []
449        solidities = []
450        areas = []
451        extents = []
452        convexities = []
453        contour_data = []
454        for det in detections:
455            x1, y1, x2, y2, score = det[:5]
456            roi = image[y1:y2, x1:x2]
457            if roi.size == 0:
458                continue
459            gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
460            contours = self._find_contours(gray)
461            for cnt in contours:
462                area = cv2.contourArea(cnt)
463                if area == 0:
464                    continue
465                perimeter = cv2.arcLength(cnt, True)
466                circularity = 4 * np.pi * area / (perimeter ** 2) if perimeter > 0 else 0
467                hull = cv2.convexHull(cnt)
468                hull_area = cv2.contourArea(hull)
469                solidity = float(area) / hull_area if hull_area > 0 else 0
470                x, y, w, h = cv2.boundingRect(cnt)
471                rect_area = w * h
472                extent = float(area) / rect_area if rect_area > 0 else 0
473                convexity = float(area) / cv2.contourArea(hull) if cv2.contourArea(hull) > 0 else 0
474                circularities.append(circularity)
475                solidities.append(solidity)
476                areas.append(area)
477                extents.append(extent)
478                convexities.append(convexity)
479                contour_data.append({
480                    "det": det,
481                    "circularity": circularity,
482                    "solidity": solidity,
483                    "area": area,
484                    "extent": extent,
485                    "convexity": convexity
486                })
487        if not contour_data:
488            return []
489        min_circ_thresh = np.percentile(circularities, 1)
490        max_circ_thresh = np.percentile(circularities, 100)
491        min_solid_thresh = np.percentile(solidities, 0)
492        max_solid_thresh = np.percentile(solidities, 100)
493        min_area_thresh = np.percentile(areas, 3)
494        max_area_thresh = np.percentile(areas, 100)
495        min_extent_thresh = np.percentile(extents, 1)
496        max_extent_thresh = np.percentile(extents, 100)
497        min_convex_thresh = np.percentile(convexities, 0)
498        max_convex_thresh = np.percentile(convexities, 100)
499        filtered_detections = []
500        det_seen = set()
501        for cdata in contour_data:
502            det = cdata["det"]
503            det_tuple = tuple(det)
504            if det_tuple in det_seen:
505                continue
506            if (0.1 <= cdata["circularity"] <= max_circ_thresh and
507                0.2 <= cdata["solidity"] <= max_solid_thresh and
508                5 <= cdata["area"] <= max_area_thresh and
509                0.1 <= cdata["extent"] <= max_extent_thresh and
510                0.1 <= cdata["convexity"] <= max_convex_thresh):
511                filtered_detections.append(det)
512                det_seen.add(det_tuple)
513        return filtered_detections

529    def _calculate_iou(self, boxA, boxB):
530        """
531        Calculates Intersection over Union (IoU) of two bounding boxes.
532
533        Args:
534            boxA (list): First box [x1, y1, x2, y2].
535            boxB (list): Second box [x1, y1, x2, y2].
536
537        Returns:
538            float: IoU value.
539        """
540        xA = max(boxA[0], boxB[0])
541        yA = max(boxA[1], boxB[1])
542        xB = min(boxA[2], boxB[2])
543        yB = min(boxA[3], boxB[3])
544        interArea = max(0, xB - xA + 1) * max(0, yB - yA + 1)
545        boxAArea = (boxA[2] - boxA[0] + 1) * (boxA[3] - boxA[1] + 1)
546        boxBArea = (boxB[2] - boxB[0] + 1) * (boxB[3] - boxB[1] + 1)
547        iou = interArea / float(boxAArea + boxBArea - interArea)
548        return iou

550    def calculate_metrics(self, all_gt_boxes, all_det_boxes, iou_threshold=0.3):
551        """
552        Calculates Precision and Recall across all images.
553
554        Args:
555            all_gt_boxes (list): List of ground truth boxes.
556            all_det_boxes (list): List of detection boxes.
557            iou_threshold (float): IoU threshold for matching.
558
559        Returns:
560            tuple: (precision, recall, true_positives, false_positives, false_negatives)
561        """
562        all_det_boxes.sort(key=lambda x: x[4], reverse=True)
563        true_positives = 0
564        false_positives = 0
565        false_negatives = 0
566        matched_gt = [False] * len(all_gt_boxes)
567        for det_idx, det_box_with_conf in enumerate(all_det_boxes):
568            det_box = det_box_with_conf[:4]
569            best_iou = 0
570            best_gt_idx = -1
571            for gt_idx, gt_box in enumerate(all_gt_boxes):
572                iou = self._calculate_iou(det_box, gt_box)
573                if iou > best_iou:
574                    best_iou = iou
575                    best_gt_idx = gt_idx
576            if best_iou >= iou_threshold:
577                if not matched_gt[best_gt_idx]:
578                    true_positives += 1
579                    matched_gt[best_gt_idx] = True
580                else:
581                    false_positives += 1 
582            else:
583                false_positives += 1
584        false_negatives = sum(1 for matched in matched_gt if not matched)
585        precision = true_positives / (true_positives + false_positives) if (true_positives + false_positives) > 0 else 0
586        recall = true_positives / (true_positives + false_negatives) if (true_positives + false_negatives) > 0 else 0
587        return precision, recall, true_positives, false_positives, false_negatives

437        """
438        Filter detections by shape metrics using percentile thresholds calculated
439        from all detected contours in the image.
440
441        Args:
442            image (np.ndarray): Input image.
443            detections (list): List of detection boxes.
444
445        Returns:
446            list: Filtered detections.
447        """
448        circularities = []
449        solidities = []
450        areas = []
451        extents = []
452        convexities = []
453        contour_data = []
454        for det in detections:
455            x1, y1, x2, y2, score = det[:5]
456            roi = image[y1:y2, x1:x2]
457            if roi.size == 0:
458                continue
459            gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
460            contours = self._find_contours(gray)
461            for cnt in contours:
462                area = cv2.contourArea(cnt)
463                if area == 0:
464                    continue
465                perimeter = cv2.arcLength(cnt, True)
466                circularity = 4 * np.pi * area / (perimeter ** 2) if perimeter > 0 else 0
467                hull = cv2.convexHull(cnt)
468                hull_area = cv2.contourArea(hull)
469                solidity = float(area) / hull_area if hull_area > 0 else 0
470                x, y, w, h = cv2.boundingRect(cnt)
471                rect_area = w * h
472                extent = float(area) / rect_area if rect_area > 0 else 0
473                convexity = float(area) / cv2.contourArea(hull) if cv2.contourArea(hull) > 0 else 0
474                circularities.append(circularity)
475                solidities.append(solidity)
476                areas.append(area)
477                extents.append(extent)
478                convexities.append(convexity)
479                contour_data.append({
480                    "det": det,
481                    "circularity": circularity,
482                    "solidity": solidity,
483                    "area": area,
484                    "extent": extent,
485                    "convexity": convexity
486                })
487        if not contour_data:
488            return []
489        min_circ_thresh = np.percentile(circularities, 1)
490        max_circ_thresh = np.percentile(circularities, 100)
491        min_solid_thresh = np.percentile(solidities, 0)
492        max_solid_thresh = np.percentile(solidities, 100)
493        min_area_thresh = np.percentile(areas, 3)
494        max_area_thresh = np.percentile(areas, 100)
495        min_extent_thresh = np.percentile(extents, 1)
496        max_extent_thresh = np.percentile(extents, 100)
497        min_convex_thresh = np.percentile(convexities, 0)
498        max_convex_thresh = np.percentile(convexities, 100)
499        filtered_detections = []
500        det_seen = set()
501        for cdata in contour_data:
502            det = cdata["det"]
503            det_tuple = tuple(det)
504            if det_tuple in det_seen:
505                continue
506            if (0.1 <= cdata["circularity"] <= max_circ_thresh and
507                0.2 <= cdata["solidity"] <= max_solid_thresh and
508                5 <= cdata["area"] <= max_area_thresh and
509                0.1 <= cdata["extent"] <= max_extent_thresh and
510                0.1 <= cdata["convexity"] <= max_convex_thresh):
511                filtered_detections.append(det)
512                det_seen.add(det_tuple)
513        return filtered_detections

551        """

553
554        Args:
555            all_gt_boxes (list): List of ground truth boxes.

557            iou_threshold (float): IoU threshold for matching.
558

560            tuple: (precision, recall, true_positives, false_positives, false_negatives)

562        all_det_boxes.sort(key=lambda x: x[4], reverse=True)

566        matched_gt = [False] * len(all_gt_boxes)

579                    matched_gt[best_gt_idx] = True

581                    false_positives += 1 
582            else:
583                false_positives += 1
584        false_negatives = sum(1 for matched in matched_gt if not matched)

586        recall = true_positives / (true_positives + false_negatives) if (true_positives + false_negatives) > 0 else 0
